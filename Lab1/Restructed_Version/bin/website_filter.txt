# 网站过滤规则配置文件
# 格式: allow/deny hostname
# 支持子串匹配，例如 "baidu.com" 会匹配 "www.baidu.com", "tieba.baidu.com" 等

# 规则说明：
# - 如果没有匹配的规则，默认允许访问
# - 规则按顺序匹配，第一个匹配的规则生效
# - 使用 # 开头的行为注释

# 示例配置
# deny baidu.com
# deny qq.com
# allow github.com

# URL路径过滤示例：
# 阻止特定页面，但允许同一网站的其他页面
# deny localhost /filter-blocked.html
# 等价于: deny localhost:80 /filter-blocked.html


deny 165.99.42.83:30000 /filter-blocked.html
